 33def download_extension(browser, url, output_dir, save_metadata=False, extract=False, show_progress=True, verify_hash=None):
 34    """
 35    Download an extension from a web store.
 36    """
 37    downloader = get_downloader(browser)
 38    if not downloader:
 39        raise ConfigError(f"Unsupported browser type: {browser}")
 40
 41    extension_id = downloader.extract_id(url)
 42    if show_progress:
 43        logger.info(f"Extracted ID/Slug: {extension_id}")
 44
 45    # Initialize hooks
 46    config_dir = get_config_path().parent
 47    hooks_dir = config_dir / "hooks"
 48    hook_manager = HookManager(hooks_dir)
 49    
 50    # Load config for hooks
 51    try:
 52        config = load_config()
 53    except Exception:
 54        config = {}
 55
 56    # Run pre-download hook
 57    ctx = HookContext(extension_id=extension_id, browser=browser, config=config)
 58    hook_manager.run_hook("pre_download", ctx)
 59    
 60    if ctx.cancel:
 61        logger.info("Download cancelled by pre_download hook.")
 62        return None
 63
 64    output_dir = Path(output_dir)
 65    if not output_dir.exists():
 66        output_dir.mkdir(parents=True, exist_ok=True)
 67
 68    output_path = downloader.download(extension_id, output_dir, show_progress=show_progress)
 69    
 70    # Verify hash if requested
 71    if verify_hash:
 72        if show_progress:
 73            logger.info(f"Verifying SHA256 hash: {verify_hash}")
 74        try:
 75            verify_file_hash(output_path, verify_hash)
 76            if show_progress:
 77                logger.info("Hash verification successful.")
 78        except IntegrityError as e:
 79            logger.error(f"Integrity check failed: {e}")
 80            # Cleanup
 81            if output_path.exists():
 82                output_path.unlink()
 83            raise
 84
 85    # Update context with result
 86    ctx.file_path = output_path
 87
 88    if save_metadata:
 89        if show_progress:
 90            logger.info("Generating metadata sidecar...")
 91        try:
 92            inspector = ExtensionInspector()
 93            manifest = inspector.get_manifest(output_path)
 94            
 95            metadata = {
 96                "id": extension_id,
 97                "name": manifest.get("name", "Unknown"),
 98                "version": manifest.get("version", "Unknown"),
 99                "source_url": url,
100                "download_timestamp": datetime.now(timezone.utc).isoformat(),
101                "filename": output_path.name
102            }
103            
104            # Add metadata to hook context
105            ctx.metadata = metadata
106            
107            # Save as <filename>.json (e.g. extension.crx.json)
108            metadata_path = output_path.with_suffix(output_path.suffix + ".json")
109            
110            with metadata_path.open("w") as f:
111                json.dump(metadata, f, indent=2)
112                
113            if show_progress:
114                logger.info(f"Metadata saved to {metadata_path}")
115        except Exception as e:
116            logger.warning(f"Failed to generate metadata: {e}")
117
118    # Run post-download hook
119    hook_manager.run_hook("post_download", ctx)
120
121    # Log to history
122    try:
123        history = HistoryManager()
124        version = None
125        if save_metadata and 'metadata' in locals():
126            version = metadata.get("version")
127        
128        history.add_entry(
129            action="download",
130            extension_id=extension_id,
131            browser=browser,
132            version=version,
133            status="success",
134            path=str(output_path)
135        )
136    except Exception as e:
137        logger.warning(f"Failed to update history: {e}")
138
139    if extract:
140        extract_extension(output_path, output_dir / output_path.stem, show_progress=show_progress)
141
142    return output_path

513    """
514    Extract an extension archive.
515    """
516    file_path = Path(file_path)
517    if not file_path.exists():
518        raise ExtensionError(f"File not found: {file_path}")
519    
520    if output_dir:
521        extract_dir = Path(output_dir)
522    else:
523        extract_dir = Path(".") / file_path.stem
524        
525    if extract_dir.exists() and any(extract_dir.iterdir()):
526         logger.warning(f"Extraction directory {extract_dir} is not empty.")
527    
528    extract_dir.mkdir(parents=True, exist_ok=True)
529    
530    if show_progress:
531        logger.info(f"Extracting {file_path} to {extract_dir}...")
532    
533    try:
534        with open_extension_archive(file_path) as zf:
535            # Check disk space
536            total_size = sum(info.file_size for info in zf.infolist())
537            check_disk_space(extract_dir, total_size)
538            
539            zf.extractall(extract_dir)
540        if show_progress:
541            logger.info(f"Successfully extracted to {extract_dir}")
542
543        # Run post-extract hook
544        try:
545            config_dir = get_config_path().parent
546            hooks_dir = config_dir / "hooks"
547            hook_manager = HookManager(hooks_dir)
548            try:
549                config = load_config()
550            except Exception:
551                config = {}
552            
553            # Try to infer ID/browser for context
554            extension_id = "unknown"
555            browser = "unknown"
556            metadata_path = file_path.with_suffix(file_path.suffix + ".json")
557            if metadata_path.exists():
558                try:
559                    with metadata_path.open("r") as f:
560                        meta = json.load(f)
561                        extension_id = meta.get("id", "unknown")
562                        source_url = meta.get("source_url", "")
563                        if "google.com" in source_url:

566                            browser = "edge"

568                            browser = "firefox"
569                except Exception:
570                    pass
571
572            ctx = HookContext(
573                extension_id=extension_id, 
574                browser=browser, 
575                file_path=extract_dir,
576                config=config
577            )
578            hook_manager.run_hook("post_extract", ctx)
579        except Exception as e:
580            logger.warning(f"Failed to run post-extract hook: {e}")
581

583        try:
584            # Try to find metadata sidecar
585            metadata_path = file_path.with_suffix(file_path.suffix + ".json")
586            extension_id = "unknown"
587            browser = "unknown"
588            version = None
589            
590            if metadata_path.exists():

592                    meta = json.load(f)
